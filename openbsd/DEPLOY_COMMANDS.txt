# OpenBSD Deployment Commands
# Run these from your terminal where SSH works:

# 1. Upload the script
scp G:\pub\openbsd\openbsd.sh dev@185.52.176.18:~/

# 2. Connect to VPS
ssh dev@185.52.176.18

# 3. On the VPS, run:
chmod +x openbsd.sh

# 4. Check if Stage 1 was completed
ls -la /var/nsd/etc/nsd.conf openbsd_setup_state 2>/dev/null

# 5. Since DNS has propagated, run Stage 2:
doas zsh openbsd.sh --resume

# The script will:
# - Verify DNS propagation
# - Configure production PF firewall
# - Configure OpenSMTPD with proper TLS
# - Prompt you to upload Rails apps
# - Deploy apps with Falcon
# - Configure relayd with TLS
# - Start all services

# After Stage 2, verify:
rcctl ls started
doas pfctl -sr
dig @185.52.176.18 brgen.no SOA +short
curl -I https://brgen.no
